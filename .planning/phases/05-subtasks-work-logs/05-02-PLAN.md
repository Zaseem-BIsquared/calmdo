       ```typescript
       /** Parse human-friendly time input to minutes. */
       export function parseTimeInput(input: string): number | undefined {
         const trimmed = input.trim();
         if (!trimmed) return undefined;

         // Match "1h30m", "1h 30m"
         const hm = trimmed.match(/^(\d+)\s*h\s*(\d+)\s*m$/i);
         if (hm) return parseInt(hm[1]) * 60 + parseInt(hm[2]);

         // Match "1.5h", "2h"
         const h = trimmed.match(/^(\d+\.?\d*)\s*h$/i);
         if (h) return Math.round(parseFloat(h[1]) * 60);

         // Match "30m"
         const m = trimmed.match(/^(\d+)\s*m$/i);
         if (m) return parseInt(m[1]);

         // Plain number = minutes
         const n = trimmed.match(/^(\d+)$/);
         if (n) return parseInt(n[1]);

         return undefined; // Invalid input
       }

       /** Format minutes to human-readable string. */
       export function formatMinutes(minutes: number | undefined): string {
         if (minutes === undefined || minutes === null) return "";
         if (minutes === 0) return "0m";
         const h = Math.floor(minutes / 60);
         const m = minutes % 60;
         if (h > 0 && m > 0) return `${h}h ${m}m`;
         if (h > 0) return `${h}h`;
         return `${m}m`;
       }
       ```

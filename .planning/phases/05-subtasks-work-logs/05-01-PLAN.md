       **Correction — use plain mutation for create** since we need `taskId` (a Convex ID) as an arg. The zCustomMutation approach only works cleanly when ALL args are Zod-validatable. Since `taskId` is a Convex ID type, use plain mutation:

       ```typescript
       import { mutation } from "@cvx/_generated/server";
       import { auth } from "@cvx/auth";
       import { v } from "convex/values";
       import { ERRORS } from "../../src/shared/errors";

       export const create = mutation({
         args: {
           taskId: v.id("tasks"),
           title: v.string(),
           assigneeId: v.optional(v.id("users")),
         },
         handler: async (ctx, args) => {
           const userId = await auth.getUserId(ctx);
           if (!userId) return;
           return await ctx.db.insert("subtasks", {
             taskId: args.taskId,
             title: args.title.trim(),
             status: "todo",
             position: Date.now(),
             assigneeId: args.assigneeId,
             createdBy: userId,
           });
         },
       });

       export const update = mutation({
         args: {
           subtaskId: v.id("subtasks"),
           title: v.optional(v.string()),
           assigneeId: v.optional(v.id("users")),
         },
         handler: async (ctx, args) => {
           const userId = await auth.getUserId(ctx);
           if (!userId) return;
           const subtask = await ctx.db.get(args.subtaskId);
           if (!subtask) throw new Error(ERRORS.subtasks.NOT_FOUND);
           const patch: Record<string, unknown> = {};
           if (args.title !== undefined) patch.title = args.title.trim();
           if (args.assigneeId !== undefined) patch.assigneeId = args.assigneeId;
           await ctx.db.patch(args.subtaskId, patch);
         },
       });

       export const remove = mutation({
         args: { subtaskId: v.id("subtasks") },
         handler: async (ctx, args) => {
           const userId = await auth.getUserId(ctx);
           if (!userId) return;
           await ctx.db.delete(args.subtaskId);
         },
       });

       export const reorder = mutation({
         args: {
           subtaskId: v.id("subtasks"),
           newPosition: v.number(),
         },
         handler: async (ctx, args) => {
           const userId = await auth.getUserId(ctx);
           if (!userId) return;
           await ctx.db.patch(args.subtaskId, { position: args.newPosition });
         },
       });

       export const promote = mutation({
         args: { subtaskId: v.id("subtasks") },
         handler: async (ctx, args) => {
           const userId = await auth.getUserId(ctx);
           if (!userId) return;
           const subtask = await ctx.db.get(args.subtaskId);
           if (!subtask) throw new Error(ERRORS.subtasks.NOT_FOUND);
           if (subtask.status === "promoted") {
             throw new Error(ERRORS.subtasks.ALREADY_PROMOTED);
           }
           // Get parent task for project inheritance
           const parentTask = await ctx.db.get(subtask.taskId);
           const newTaskId = await ctx.db.insert("tasks", {
             title: subtask.title,
             status: "todo",
             visibility: parentTask?.projectId ? "shared" : "private",
             priority: false,
             creatorId: userId,
             assigneeId: userId,
             projectId: parentTask?.projectId,
             position: Date.now(),
           });
           await ctx.db.patch(args.subtaskId, {
             status: "promoted",
             promotedToTaskId: newTaskId,
           });
           return newTaskId;
         },
       });
       ```

       ```typescript
       import { query } from "@cvx/_generated/server";
       import { auth } from "@cvx/auth";
       import { v } from "convex/values";

       export const listByTask = query({
         args: { taskId: v.id("tasks") },
         handler: async (ctx, args) => {
           const userId = await auth.getUserId(ctx);
           if (!userId) return { subtasks: [], completionCount: { done: 0, total: 0 } };
           const subtasks = await ctx.db
             .query("subtasks")
             .withIndex("by_task", (q) => q.eq("taskId", args.taskId))
             .collect();
           const sorted = subtasks.sort((a, b) => a.position - b.position);
           const total = sorted.length;
           const done = sorted.filter((s) => s.status === "done").length;
           return { subtasks: sorted, completionCount: { done, total } };
         },
       });
       ```

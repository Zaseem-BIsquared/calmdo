       Create `assignToProject` (plain mutation):
       ```typescript
       export const assignToProject = mutation({
         args: {
           taskId: v.id("tasks"),
           projectId: v.optional(v.id("projects")),
         },
         handler: async (ctx, args) => {
           const userId = await auth.getUserId(ctx);
           if (!userId) return;
           const task = await ctx.db.get(args.taskId);
           if (!task) throw new Error(ERRORS.tasks.NOT_FOUND);

           const patch: Record<string, unknown> = { projectId: args.projectId };
           if (args.projectId) {
             // Moving to project: auto-flip to shared (per D-06)
             patch.visibility = "shared";
           } else {
             // Removing from project: restore private if creator=assignee (per D-07)
             if (task.assigneeId === task.creatorId || !task.assigneeId) {
               patch.visibility = task.assigneeId === task.creatorId ? "private" : "shared";
             }
             // else: stays shared (assigned to someone other than creator)
           }
           await ctx.db.patch(args.taskId, patch);
         },
       });
       ```

       Also extend `create` to accept an optional projectId for inline task creation on project detail page (per D-08). Since `create` uses zMutation with `createTaskInput`, and we don't want to pollute the Zod schema: override the create mutation to use plain `mutation` instead, OR add a `createInProject` mutation. **Go with `createInProject`:**

       export const createInProject = mutation({
         args: {
